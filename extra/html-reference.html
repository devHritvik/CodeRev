Form Elements

Registration Form HTML5

A complete registration form with various input types.

<form action="/register" method="post">
  <fieldset>
    <legend>Personal Information</legend>
    <div>
      <label for="name">Full Name:</label>
      <input type="text" id="name" name="name" required>
    </div>

    <div>
      <label for="email">Email:</label>
      <input type="email" id="email" name="email" required>
    </div>

    <div>
      <label for="birthdate">Birth Date:</label>
      <input type="date" id="birthdate" name="birthdate">
    </div>

    <div>
      <label for="color">Favorite Color:</label>
      <input type="color" id="color" name="color">
    </div>

    <div>
      <label for="range">Experience Level:</label>
      <input type="range" id="range" name="range" min="0" max="10">
    </div>
  </fieldset>

  <button type="submit">Register</button>
  <button type="reset">Clear Form</button>
</form>

Canvas HTML5

Used to draw graphics via JavaScript.

<canvas id="myCanvas" width="200" height="200"></canvas>
<script>
  const canvas = document.getElementById('myCanvas');
  const ctx = canvas.getContext('2d');
  ctx.fillStyle = 'red';
  ctx.fillRect(10, 10, 100, 100);
  ctx.fillStyle = 'blue';
  ctx.beginPath();
  ctx.arc(150, 150, 30, 0, 2 * Math.PI);
  ctx.fill();
</script>
